package client;

import org.springframework.boot.ApplicationArguments;
import org.springframework.boot.ApplicationRunner;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.core.ParameterizedTypeReference;
import org.springframework.web.bind.annotation.*;

import robot.HumanBot;
import robot.Robot;
import org.springframework.web.client.RestClient;

import java.util.*;

/**
 * Spring Boot client application that:
 *  1. Exposes GET /action  — called by RemoteBot on the server during tournament play.
 *     The response is the local bot's chosen action (blocks on stdin for HumanBot).
 *
 *  2. On startup (after Tomcat is ready), guides the user through:
 *     - Fetching available tournaments from the server
 *     - Picking a tournament and registering
 *
 * Run with --spring.profiles.active=client to use port 8081 (see application-client.properties).
 */
@SpringBootApplication
@RestController
public class NetworkedTournamentClient implements ApplicationRunner {

    // The local bot — set after the user decides their strategy (human or algorithmic)
    private Robot bot;

    // Intentionally not closed: closing a Scanner wrapping System.in would close stdin
    private final Scanner console = new Scanner(System.in);

    public static void main(String[] args) {
        SpringApplication.run(NetworkedTournamentClient.class, args);
    }

    /**
     * GET /action
     * Called by the server's RemoteBot during each round of the tournament.
     * If bot is a HumanBot, this blocks until the human types a response.
     */
    @GetMapping("/action")
    public String getAction() {
        if (bot == null) {
            return "COOPERATE"; // safe default before registration completes
        }
        return bot.getAction();
    }

    /**
     * Runs after the embedded Tomcat is fully started,
     * so /action is reachable before the user finishes the registration dialog.
     */
    @Override
    public void run(ApplicationArguments args) throws Exception {
        System.out.print("Enter server IP:port (e.g. 192.168.1.10:8080): ");
        String serverIp = console.nextLine().trim();

        RestClient client = RestClient.create();

        // Fetch available tournaments
        List<Map<String, Object>> tournamentList = client.get()
            .uri("http://" + serverIp + "/tournaments")
            .retrieve()
            .body(new ParameterizedTypeReference<>() {});

        System.out.println("\nAvailable Tournaments:");
        for (int i = 0; i < tournamentList.size(); i++) {
            Map<String, Object> t = tournamentList.get(i);
            String status = Boolean.TRUE.equals(t.get("open")) ? "OPEN" : "CLOSED";
            System.out.println("  " + (i + 1) + ". " + t.get("name") + " [" + status + "]");
        }

        System.out.print("Pick tournament number: ");
        int pick = Integer.parseInt(console.nextLine().trim()) - 1;
        String tournamentName = (String) tournamentList.get(pick).get("name");

        System.out.print("Enter your robot name: ");
        String robotName = console.nextLine().trim();

        System.out.print("Are you a human player? (y/n): ");
        boolean isHuman = console.nextLine().trim().equalsIgnoreCase("y");

        // Create the local bot
        bot = new HumanBot(robotName);

        // For human players, no callback IP needed (server creates HumanBot locally)
        // For remote bots, provide this machine's IP:port so the server can call /action
        String myIp = "";
        if (!isHuman) {
            System.out.print("Enter this machine's IP:port (e.g. 192.168.1.20:8081): ");
            myIp = console.nextLine().trim();
        }

        // Register with the server
        Map<String, String> regBody = new HashMap<>();
        regBody.put("robotName", robotName);
        regBody.put("tournamentName", tournamentName);
        regBody.put("ip", myIp);

        Boolean success = client.post()
            .uri("http://" + serverIp + "/register")
            .contentType(org.springframework.http.MediaType.APPLICATION_JSON)
            .body(regBody)
            .retrieve()
            .body(Boolean.class);

        if (Boolean.TRUE.equals(success)) {
            System.out.println("\nRegistered successfully in \"" + tournamentName + "\"!");
            if (!isHuman) {
                System.out.println("Listening for move requests on /action ...");
                System.out.println("(The server will call this endpoint each round of the game)");
            } else {
                System.out.println("You will be prompted here when the tournament starts.");
            }
        } else {
            System.out.println("Registration failed. Tournament may be full or not found.");
        }
    }
}

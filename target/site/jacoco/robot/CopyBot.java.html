package robot;

// A robot that cooperates on the first move, then copies the opponent's last action.
public class CopyBot extends Robot implements OpponentAware {
    private int score;
    private String currentOpponent;

    public CopyBot(String name) {
        super(name);
        this.score = 0;
    }

    @Override
    public void setCurrentOpponent(String opponentName) {
        this.currentOpponent = opponentName;
    }

    @Override
    public String getAction() {
        if (currentOpponent == null) {
            return "COOPERATE";
        }
        String lastAction = getLastOpponentAction(currentOpponent);
        return (lastAction == null) ? "COOPERATE" : lastAction;
    }

    @Override
    public int getScore() {
        return score;
    }

    @Override
    public void giveOutcome(String opponentName, String myAction,
                            String opponentAction, int myScore) {
        super.giveOutcome(opponentName, myAction, opponentAction, myScore);
        this.score += myScore;
    }
}
